Let $(X, d_X)$ and $(Y, d_Y)$ be two metric spaces.

<Theorem title="isometry">
$\phi: X\to Y$ is an isometry if 

$$
(\forall\; a,b\in X)\;d_X(a,b)=d_Y(\phi(a), \phi(b))
$$
</Theorem>

Two examples:


export const colors = ["#F5BD1E", "#5F9C3F", "#6BD1B9"]; 
export const IsometryGraph = () => {
  const nodes = [
    { x: 200, y: 50 },  // Top
    { x: 100, y: 150 }, // Bottom Left
    { x: 300, y: 150 }, // Bottom Right
  ];

  const [order, setOrder] = useState([0, 1, 2]);

  useEffect(() => {
    const interval = setInterval(() => {
      setOrder((prev) => [prev[2], prev[0], prev[1]]); 
    }, 2000); // Change every 2 seconds
    return () => clearInterval(interval);
  }, []);

  return (
    <svg width="400" height="250" style={{ overflow: 'visible', margin: 'auto', display: 'block' }}>
      {/* Static Edges (The "Metric Space" doesn't move) */}
      <line x1={nodes[0].x} y1={nodes[0].y} x2={nodes[1].x} y2={nodes[1].y} stroke="gray" strokeWidth="2" strokeDasharray="4" />
      <line x1={nodes[1].x} y1={nodes[1].y} x2={nodes[2].x} y2={nodes[2].y} stroke="gray" strokeWidth="2" strokeDasharray="4" />
      <line x1={nodes[2].x} y1={nodes[2].y} x2={nodes[0].x} y2={nodes[0].y} stroke="gray" strokeWidth="2" strokeDasharray="4" />

      {/* Animated Vertices */}
      {colors.map((color, i) => (
        <motion.circle
          key={color}
          r="12"
          fill={color}
          animate={{ 
            cx: nodes[order[i]].x, 
            cy: nodes[order[i]].y 
          }}
          transition={{ 
            type: "spring", 
            stiffness: 60, 
            damping: 10 
          }}
        />
      ))}
    </svg>
  );
};
export const ReflectionGraph = () => {
  const nodes = [
    { x: 200, y: 50 },  // Index 0: Top (The Mirror Axis)
    { x: 100, y: 150 }, // Index 1: Bottom Left
    { x: 300, y: 150 }, // Index 2: Bottom Right
  ];

  const [order, setOrder] = useState([0, 1, 2]);

  useEffect(() => {
    const interval = setInterval(() => {
      setOrder((prev) => (prev[1] === 1 ? [0, 2, 1] : [0, 1, 2]));
    }, 1500); 
    return () => clearInterval(interval);
  }, []);

  return (
    <svg width="400" height="250" style={{ overflow: 'visible', margin: 'auto', display: 'block' }}>
      {/* Mirror Axis Line (Optional visual aid) */}
      <line x1="200" y1="20" x2="200" y2="180" stroke="#444" strokeDasharray="5,5" />

      <line x1={nodes[0].x} y1={nodes[0].y} x2={nodes[1].x} y2={nodes[1].y} stroke="gray" strokeWidth="2" />
      <line x1={nodes[1].x} y1={nodes[1].y} x2={nodes[2].x} y2={nodes[2].y} stroke="gray" strokeWidth="2" />
      <line x1={nodes[2].x} y1={nodes[2].y} x2={nodes[0].x} y2={nodes[0].y} stroke="gray" strokeWidth="2" />

      {colors.map((color, i) => (
        <motion.circle
          key={color}
          r="12"
          fill={color}
          animate={{ 
            cx: nodes[order[i]].x, 
            cy: nodes[order[i]].y 
          }}
          transition={{ 
            type: "spring", 
            stiffness: 40, 
            damping: 12 
          }}
        />
      ))}
    </svg>
  );
};
